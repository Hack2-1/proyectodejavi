# crontab.txt - ejemplos de programacion con cron
# para usarlo abre: crontab -e
# estas rutas son para la instalacion sin root que hace setup.sh

SHELL=/bin/bash

# respaldo diario a las 02:00
0 2 * * * $HOME/.local/share/sistema-servicios/respaldo.sh >> $HOME/.local/state/sistema-servicios/logs/cron-respaldo.log 2>&1

# monitoreo cada 5 minutos
*/5 * * * * $HOME/.local/share/sistema-servicios/monitoreo.sh >> $HOME/.local/state/sistema-servicios/logs/cron-monitoreo.log 2>&1

# revision de servicios cada 10 minutos
*/10 * * * * $HOME/.local/share/sistema-servicios/servicios.sh >> $HOME/.local/state/sistema-servicios/logs/cron-servicios.log 2>&1

# monitoreo de red cada 15 minutos
*/15 * * * * $HOME/.local/share/sistema-servicios/red.sh >> $HOME/.local/state/sistema-servicios/logs/cron-red.log 2>&1

# inventario diario a las 00:30
30 0 * * * $HOME/.local/share/sistema-servicios/inventario.sh >> $HOME/.local/state/sistema-servicios/logs/cron-inventario.log 2>&1

# ejecucion remota de diagnostico todos los dias a las 02:30
30 2 * * * $HOME/.local/share/sistema-servicios/remoto.sh $HOME/.local/share/sistema-servicios/diagnostico.sh $HOME/.local/share/sistema-servicios/hosts.txt >> $HOME/.local/state/sistema-servicios/logs/cron-remoto.log 2>&1

# formato cron:
# minuto hora dia_mes mes dia_semana comando
# ejemplo: */5 * * * * significa cada 5 minutos
